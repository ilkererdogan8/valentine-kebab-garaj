    <script>
        // Create swamp bubbles
        function createBubbles() {
            const container = document.getElementById('swampParticles');
            for (let i = 0; i < 20; i++) {
                const bubble = document.createElement('div');
                bubble.className = 'bubble';
                const size = Math.random() * 30 + 10;
                bubble.style.width = size + 'px';
                bubble.style.height = size + 'px';
                bubble.style.left = Math.random() * 100 + '%';
                bubble.style.animationDuration = (Math.random() * 10 + 8) + 's';
                bubble.style.animationDelay = (Math.random() * 10) + 's';
                container.appendChild(bubble);
            }
        }

        // Create floating hearts
        function createFloatingHearts() {
            const container = document.getElementById('floatingHearts');
            const hearts = ['💚', '🧡', '💛', '🤎', '💚', '🧅'];
            for (let i = 0; i < 15; i++) {
                const heart = document.createElement('div');
                heart.className = 'float-heart';
                heart.textContent = hearts[Math.floor(Math.random() * hearts.length)];
                heart.style.left = Math.random() * 100 + '%';
                heart.style.animationDuration = (Math.random() * 15 + 10) + 's';
                heart.style.animationDelay = (Math.random() * 15) + 's';
                heart.style.fontSize = (Math.random() * 1.5 + 0.8) + 'rem';
                container.appendChild(heart);
            }
        }

        // Create fireflies
        function createFireflies() {
            const container = document.getElementById('fireflies');
            for (let i = 0; i < 12; i++) {
                const firefly = document.createElement('div');
                firefly.className = 'firefly';
                firefly.style.left = Math.random() * 100 + '%';
                firefly.style.top = Math.random() * 100 + '%';
                firefly.style.animationDuration = (Math.random() * 6 + 5) + 's';
                firefly.style.animationDelay = (Math.random() * 8) + 's';
                container.appendChild(firefly);
            }
        }

        // Create stars
        function createStars() {
            const container = document.getElementById('stars');
            for (let i = 0; i < 50; i++) {
                const star = document.createElement('div');
                star.className = 'star';
                star.style.left = Math.random() * 100 + '%';
                star.style.top = Math.random() * 60 + '%';
                star.style.animationDelay = (Math.random() * 3) + 's';
                star.style.animationDuration = (Math.random() * 2 + 2) + 's';
                container.appendChild(star);
            }
        }

        // Scroll animations
        function handleScrollAnimations() {
            const elements = document.querySelectorAll('.fade-in');
            elements.forEach(el => {
                const rect = el.getBoundingClientRect();
                const windowHeight = window.innerHeight;
                if (rect.top < windowHeight * 0.85) {
                    el.classList.add('visible');
                }
            });
        }

        // Music toggle
        let musicPlaying = false;
        function toggleMusic() {
            const btn = document.getElementById('musicBtn');
            musicPlaying = !musicPlaying;
            btn.textContent = musicPlaying ? '🔊' : '🎵';
            btn.style.background = musicPlaying ? 'rgba(144, 238, 144, 0.3)' : 'rgba(34, 80, 20, 0.8)';
        }

        // Remove entrance overlay
        setTimeout(() => {
            const entrance = document.getElementById('entrance');
            if (entrance) {
                entrance.style.display = 'none';
            }
        }, 4000);

        // Initialize
        document.addEventListener('DOMContentLoaded', () => {
            createBubbles();
            createFloatingHearts();
            createFireflies();
            createStars();
            handleScrollAnimations();
        });

        window.addEventListener('scroll', handleScrollAnimations);

        // Easter egg: Click on characters for messages
        document.querySelectorAll('.shrek-character').forEach(el => {
            el.addEventListener('click', () => {
                alert("🧅 GET OUT OF MY SWAMP!\n\n...just kidding, come closer, Agata 💚\n\n— Ilker");
            });
        });

        document.querySelectorAll('.fiona-character').forEach(el => {
            el.addEventListener('click', () => {
                alert("👑 Every princess deserves her ogre!\n\n💚 I love you, Ilker!\n\n— Agata");
            });
        });

        // Confetti on double-click anywhere
        document.addEventListener('dblclick', (e) => {
            for (let i = 0; i < 20; i++) {
                const confetti = document.createElement('div');
                confetti.textContent = ['💚', '🧅', '👑', '🧡', '✨'][Math.floor(Math.random() * 5)];
                confetti.style.position = 'fixed';
                confetti.style.left = e.clientX + 'px';
                confetti.style.top = e.clientY + 'px';
                confetti.style.fontSize = '1.5rem';
                confetti.style.pointerEvents = 'none';
                confetti.style.zIndex = '9999';
                confetti.style.transition = 'all 1.5s ease-out';
                document.body.appendChild(confetti);

                setTimeout(() => {
                    confetti.style.left = (e.clientX + (Math.random() - 0.5) * 300) + 'px';
                    confetti.style.top = (e.clientY - Math.random() * 400) + 'px';

                    confetti.style.transform = `rotate(${Math.random() * 720}deg)`;
                }, 10);

            }
